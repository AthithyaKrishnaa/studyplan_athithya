import java.util.*;

// singleton pattern
class Main{
    public static void main(String[] args){
        Printer p1 = Printer.getInstance();
        Printer p2 = Printer.getInstance();
        System.out.println(p1);
        System.out.println(p2);
    }
}
class Printer{
    private static Printer obj = null;
    private Printer(){
        
    }
    static Printer getInstance(){
        if(obj==null){
            obj = new Printer();
        }
        return obj;
    }
}
